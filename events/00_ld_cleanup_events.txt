namespace = ld

# Death cleanup - most critical for desyncs
ld.100 = {
	type = character_event
	hidden = yes
	immediate = {
		debug_log = "LD: Death cleanup fired"
		if = {
			limit = {
				is_betrothed = yes
				betrothed = { is_alive = yes }
			}
			debug_log = "LD: Clearing betrothal from dead character"
			break_betrothal = betrothed
		}
		save_scope_value_as = {
			name = cleanup_death
			value = yes
		}
	}
}

# Marriage validation - dead spouses
ld.117 = {
	type = character_event
	hidden = yes
	immediate = {
		debug_log = "LD: Marriage validation fired"
		if = {
			limit = {
				is_married = yes
				any_spouse = { is_alive = no }
			}
			debug_log = "LD: Divorcing dead spouse"
			every_spouse = {
				limit = { is_alive = no }
				ROOT = { divorce = prev }
			}
		}
	}
}
