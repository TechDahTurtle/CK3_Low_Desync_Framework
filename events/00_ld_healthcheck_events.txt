namespace = ld

game_event = {
    id = ld.1100
    hide_window = yes
    is_triggered_only = yes

    immediate = {
        if = { limit = { exists = global_var:ld_healthcheck_done }}
        set_global_variable = { name = ld_healthcheck_done value = 1 }

        if = {
            limit = {
                NOT = {
                    OR = {
                        has_game_rule = ld_mp_safe
                        has_game_rule = ld_full
                        # Add future game rule options here
                    }
                }
            }
            send_interface_toast = {
                type = event_toast_effect_bad
                title = ld_healthcheck_title
                desc = ld_healthcheck_desc
            }
        }
    }
}
