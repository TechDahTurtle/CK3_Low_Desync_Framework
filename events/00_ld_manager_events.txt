namespace = ld

character_event = {
    id = ld.1000
    hide_window = yes
    is_triggered_only = yes

    immediate = {
        if = {
            limit = { has_character_flag = ld_queue_cleanup_needed }

            remove_character_flag = ld_queue_cleanup_needed

            every_scheme = {
                limit = {
                    scheme_target_character = {
                        OR = { exists = no is_alive = no }
                    }
                }
                end_scheme = yes
            }
        }
    }
}
